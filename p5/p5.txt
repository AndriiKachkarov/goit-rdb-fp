USE pandemic;

DROP FUNCTION IF EXISTS GetYearDiff;

DELIMITER //

CREATE FUNCTION GetYearDiff(input_year INT) 
RETURNS INT
DETERMINISTIC
BEGIN
    DECLARE result_years INT;
    SET result_years = TIMESTAMPDIFF(YEAR, STR_TO_DATE(CONCAT(input_year, '-01-01'), '%Y-%m-%d'), CURDATE());
    RETURN result_years;
END //

DELIMITER ;

SELECT 
    Year, 
    GetYearDiff(Year) AS years_since_start
FROM cases_data;
